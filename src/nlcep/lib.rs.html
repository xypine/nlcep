13//! ### Usage
14//! The main logic can be accessed by constructing a [`NewEvent`] from a string. For example:
15//! ```rust
16//! # use jiff::{ Zoned, ToSpan }; // nlcep uses jiff for storing dates
17//! # let tomorrow = Zoned::now().checked_add(1.day()).unwrap();
18//! #
19//! // Parse event
20//! let event: nlcep::NewEvent =
21//!     "Meeting about Q3 duckling quotas tomorrow 11:00, A769"
22//!     .parse()
23//!     .expect("Parsing event failed");
24//!
25//! // Check that the details have been parsed correctly :)
26//! assert_eq!(event.summary, "Meeting about Q3 duckling quotas");
27//! assert_eq!(event.location, Some("A769".to_owned()));
28//! assert_eq!(event.date.year(), tomorrow.year());
29//! assert_eq!(event.date.day(), tomorrow.day());
30//! assert_eq!(event.date.month(), tomorrow.month());
31//! let time = event.time.unwrap();
32//! assert_eq!(time.hour(), 11);
33//! assert_eq!(time.minute(), 0);
34//! ```
35//!
36//! You can also use [`NewEvent::parse_at_time`] to supply an alternative basis for relative time
37//! formats such as "tomorrow" or "yesterday":
38//! ```rust
39//! use jiff::{ Timestamp, Zoned };
40//! let now: Zoned = "2024-07-11T13:14:00Z"
41//!     .parse::<Timestamp>().expect("Invalid timestamp")
42//!     .in_tz("America/New_York").expect("Invalid timezone");
43//! // Parse event with an alternative "now"
44//! let event =
45//!     nlcep::NewEvent::parse_at_time("water the plants tomorrow", now)
46//!     .expect("Parsing event failed");
47//! assert_eq!(event.date.year(), 2024);
48//! assert_eq!(event.date.month(), 7);
49//! assert_eq!(event.date.day(), 11 + 1);
50//! ```

109/// Represents a parsed event
110#[derive(Debug, Serialize, Deserialize)]
111#[cfg_attr(feature = "wasm", derive(tsify::Tsify))]
112#[cfg_attr(feature = "wasm", tsify(into_wasm_abi, from_wasm_abi))]
113pub struct NewEvent {
114    /// Summary of the parsed event
115    pub summary: String,
116    pub date: Date,
117    pub time: Option<Time>,
118    /// Where the event takes place, not mandatory
119    pub location: Option<String>,
120    /// For how long the event goes on, not mandatory
121    pub duration: Option<Span>,
122}
123
124impl PartialEq for NewEvent {
125    fn eq(&self, other: &Self) -> bool {
126        let duration_same = match (self.duration, other.duration) {
127            (None, None) => true,
128            (Some(_), None) => false,
129            (None, Some(_)) => false,
130            (Some(a), Some(b)) => a
131                .compare(b)
132                .map(|ord| matches!(ord, std::cmp::Ordering::Equal))
133                .unwrap_or(false),
134        };
135        self.summary == other.summary
136            && self.date == other.date
137            && self.time == other.time
138            && self.location == other.location
139            && duration_same
140    }
141}
142
143impl NewEvent {
144    pub fn parse_at_time(s: &str, now: Zoned) -> Result<Self, EventParseError> {
145        let mut summary: Option<String> = None;
146        let mut location: Option<String> = None;
147        let DateTimeMatch {
148            date,
149            time,
150            start_char: time_starts,
151            end_char: time_ends,
152        } = find_datetime(s, now, false)?.ok_or(EventParseError::MissingTime)?;
153        let (before_time, _) = s.split_at(time_starts);
154        let (_, after_time) = s.split_at(time_ends);
155
156        let before_time_trimmed = before_time.trim();
157        if !before_time_trimmed.is_empty() {
158            summary = Some(before_time_trimmed.to_owned());
159        }
160
161        let location_start_pattern = regex!(r"\s*[@ | ,]\s+.+");
162        if location_start_pattern.is_match(after_time) {
163            let trimmed_location = after_time
164                .trim()
165                .trim_start_matches(['@', ','])
166                .trim_start();
167            location = Some(trimmed_location.to_owned());
168        }
169
170        Ok(Self {
171            summary: summary.ok_or(EventParseError::MissingSummary)?,
172            date,
173            time,
174            location,
175            duration: None,
176        })
177    }
178
179    pub fn datetime(&self) -> DateTime {
180        self.time
181            .map_or_else(|| self.date.into(), |time| self.date.to_datetime(time))
182    }
183}

185/// Contains all possible error variants that may occur while parsing a new event.
186#[derive(Debug, PartialEq, Clone, Copy, thiserror::Error, Serialize, Deserialize)]
187#[cfg_attr(feature = "wasm", derive(tsify::Tsify))]
188#[cfg_attr(feature = "wasm", tsify(into_wasm_abi, from_wasm_abi))]
189pub enum EventParseError {
190    /// No valid datetime could be parsed, other details might be valid.
191    /// For example:
192    /// ```rust
193    /// use nlcep::{ NewEvent, EventParseError };
194    /// let err = "Meet Saara @ Local Library".parse::<NewEvent>();
195    /// assert_eq!(err, Err(EventParseError::MissingTime));
196    /// ```
197    #[error("Missing time")]
198    MissingTime,
199    /// Reserved for future use
200    #[error("Invalid time")]
201    InvalidTime,
202    /// Reserved for future use
203    #[error("Ambiguous time")]
204    AmbiguousTime,
205    /// The event contains a valid time, but a summary couldn't be found.
206    /// For example:
207    /// ```rust
208    /// use nlcep::{ NewEvent, EventParseError };
209    /// let err = "tomorrow 11:00".parse::<NewEvent>();
210    /// assert_eq!(err, Err(EventParseError::MissingSummary));
211    /// ```
212    #[error("Missing summary")]
213    MissingSummary,
214    /// Reserved for future use
215    #[error("Ambiguous duration")]
216    AmbiguousDuration,
217}
218impl FromStr for NewEvent {
219    type Err = EventParseError;
220
221    fn from_str(s: &str) -> Result<Self, Self::Err> {
222        let now = Zoned::now();
223        Self::parse_at_time(s, now)
224    }
225}
